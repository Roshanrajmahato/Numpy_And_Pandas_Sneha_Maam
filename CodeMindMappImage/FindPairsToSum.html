l1 = [2, 3, 1, 4, 2, 5, 6, 7, 1, 2, 4, 8, 1, 4, 5]

target = int(input("Enter a value: "))

seen = set()
res = []

for num in l1:

    complement = target - num

    if complement in seen:
        res.append((complement, num))

    seen.add(num)

print(res)
