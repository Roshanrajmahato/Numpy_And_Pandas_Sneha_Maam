def getFirst(v, target):
    n = len(v)
    lo = 0
    hi = n - 1
    ans = -1

    while lo <= hi:
        # mid = lo + (hi - lo) / 2
        mid = lo + (hi - lo) // 2

        if v[mid] == target:
            ans = mid
            hi = mid - 1  # Look on the left side for an earlier occurrence
        elif v[mid] < target:
            lo = mid + 1
        else:
            hi = mid - 1

    return ans


def getSecond(v, target):
    n = len(v)
    lo = 0
    hi = n - 1
    ans = -1

    while lo <= hi:
        mid = lo + (hi - lo) // 2

        if v[mid] == target:
            ans = mid
            lo = mid + 1  # Look on the right side for a later occurrence
        elif v[mid] < target:
            lo = mid + 1
        else:
            hi = mid - 1

    return ans


n = int(input("Enter n: "))

v = []
print(f"Enter {n} sorted numbers:")

target = int(input("Enter target: "))
