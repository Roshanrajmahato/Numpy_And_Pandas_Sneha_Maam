L = [10, 20, 30, 20, 10]

left_idx  = 0
right_idx = len(L) - 1

while left_idx < right_idx:

    if L[left_idx] != L[right_idx]:
        print("List is not palindrome")
        break

    left_idx  += 1
    right_idx -= 1

else:
    print("List is palindrome")
