def getNoOfDigits(n):
    digits = 0
    if n == 0:
        return 1
    temp = n
    while temp > 0:
        digits += 1
        temp //= 10
    return digits


def getSumOfDigits(n):
    sum = 0
    temp = n
    while temp > 0:
        sum += temp % 10
        temp //= 10
    return sum


def getEvenDigits(n):
    evenDigits = 0
    temp = n
    while temp > 0:
        rem = temp % 10
        if rem % 2 == 0:
            evenDigits += 1
        temp //= 10
    return evenDigits


def check(n):
    originalDigits = getNoOfDigits(n)
    st = set()
    temp = n
    while temp > 0:
        rem = temp % 10
        st.add(rem)
        temp //= 10

    if len(st) == originalDigits:
        return True
    else:
        return False


if __name__ == "__main__":
    n = int(input())

    if (
        check(n) == False
        or getNoOfDigits(n) != 4
        or getSumOfDigits(n) % 3 != 0
        or getEvenDigits(n) != 2
