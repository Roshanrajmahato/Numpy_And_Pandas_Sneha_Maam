def is_prime(n):
    if n < 2:
        return False
    for i in range(2, int(n**0.5)+1):
        if n % i == 0:
            return False
    return True

n = int(input("Enter number: "))
primes = []

for i in range(2, n):
    if is_prime(i) and n % i == 0:
        primes.append(i)

if len(primes) == 3 and primes[0]*primes[1]*primes[2] == n:
    print("Sphenic Number")
else:
    print("Not Sphenic Number")
